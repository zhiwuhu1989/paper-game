updatePlayerContainer(paperId, gridX, gridY) {
  const paper = this.papers[paperId];
  const targetContainer = paper.getPlayerContainer(gridX, gridY);
  
  if (this.player.parentContainer !== targetContainer) {
    // 世界坐标 → 局部坐标转换
    const worldX = this.player.x;
    const worldY = this.player.y;
    const localPos = targetContainer.getWorldTransformMatrix().applyInverse(worldX, worldY);
    
    // 移动玩家到新容器
    this.player.parentContainer.remove(this.player);
    targetContainer.add(this.player);
    this.player.setPosition(localPos.x, localPos.y);
  }
}
